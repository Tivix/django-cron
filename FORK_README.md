```python

```

```python
from django_cron import CronJobBase, Schedule

class ExampleCronJob(CronJobBase):
    RUN_AT_TIMES = ['10:10', '22:10']
    schedule = Schedule(run_at_times=RUN_AT_TIMES, day_of_week='2')
    code = 'cron.ExampleCronJob'
    def do(self):
        pass
```
